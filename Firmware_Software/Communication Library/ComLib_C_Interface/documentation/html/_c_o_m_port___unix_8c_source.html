57 uint32_t com_get_port_list(char* port_list, size_t buffer_size)
58 {
59 #if defined __APPLE__
60  /*
61  * On Mac devices are represented as files in /dev, search for the Com
62  * ports (cu) that have a USB in its name
63  * Note: use cu* instead of tty* because tty blocks on open for carrier
64  * detection
65  */
66  const char dev_dir[] = "/dev";
67  const char search_pattern[] = "cu.usb";
68 
69 #else
70  /*
71  * Modern Linux kernel list serial devices in /dev/serial. This is a nice
72  * feature because it can be search for IDs containing Infineon ID.
73  */
74  const char dev_dir[] = "/dev/serial/by-id";
75  const char search_pattern[] = "IFX_CDC";
76 
77 #endif
78 
79  uint32_t num_available_ports = 0;
80  DIR* dir_handle;
81 
82  /* init port list to empty string */
83  if (buffer_size > 0)
84  {
85  port_list[0] = 0;
86  }
87 
88  /* open directory that contains the serial device files */
89  dir_handle = opendir(dev_dir);
90 
91  if (dir_handle != NULL)
92  {
93  /* read entry per entry */
94  struct dirent* dir_entry;
95 
96  while ((dir_entry = readdir(dir_handle)) != NULL)
97  {
98  size_t name_length;
99 
100  /* compare device file name with the search pattern */
101  char current_device[128];
102 
103  if (strstr(dir_entry->d_name, search_pattern) == NULL)
104  {
105  continue;
106  }
107 
108  /* add the port name to the list, if buffer capacity is
109  * sufficient
110  */
111  strcpy(current_device, dev_dir);
112  strcat(current_device, "/");
113  strcat(current_device, dir_entry->d_name);
114 
115  name_length = strlen(current_device);
116 
117  if (num_available_ports > 0)
118  {
119  /* add one for the separator */
120  ++name_length;
121  }
122 
123  if (name_length < buffer_size)
124  {
125  if (num_available_ports > 0)
126  {
127  strcat(port_list, ";");
128  }
129 
130  strcat(port_list, current_device);
131 
132  /* update remaining capacity of string buffer */
133  buffer_size -= name_length;
134  }
135 
136  /* port is available, so increase counter */
137  ++num_available_ports;
138  }
139 
140  /* close directory stream */
141  closedir(dir_handle);
142  }
143 
144  return num_available_ports;
145 }

147 int32_t com_open(const char* port_name)
148 {
149  int com_port_handle;
150  struct termios options;
151  int32_t new_handle;
152 
153  /*
154  * Open the serial port read/write, with no controlling terminal, and
155  * don't wait for a connection. See open(2) ("man 2 open") for details.
156  */
157  com_port_handle = open(port_name, O_RDWR | O_NOCTTY);
158 
159  if (com_port_handle == -1)
160  {
161  return -1;
162  }
163 
164  /*
165  * Note that open() follows POSIX semantics: multiple open() calls to the
166  * same file will succeed unless the TIOCEXCL ioctl is issued. This will
167  * prevent additional opens except by root-owned processes.
168  * See tty(4) ("man 4 tty") and ioctl(2) ("man 2 ioctl") for details.
169  */
170  if (ioctl(com_port_handle, TIOCEXCL) == -1)
171  {
172  close(com_port_handle);
173  return -1;
174  }
175 
176  /* Get the current options and save them so we can restore the default
177  * settings later.
178  */
179  if (tcgetattr(com_port_handle, &options) == -1)
180  {
181  close(com_port_handle);
182  return -1;
183  }
184 
185  /*
186  * The serial port attributes such as timeouts and baud rate are set by
187  * modifying the termios structure and then calling tcsetattr to cause
188  * the changes to take effect. Note that the changes will not take effect
189  * without the tcsetattr() call. See tcsetattr(4) ("man 4 tcsetattr") for
190  * details.
191  *
192  * Set raw input (non-canonical) mode, with reads blocking until either a
193  * single character has been received or a one second timeout expires.
194  * See tcsetattr(4) ("man 4 tcsetattr") and termios(4) ("man 4 termios")
195  * for details.
196  */
197  cfmakeraw(&options);
198  options.c_cc[VMIN] = 0;
199  options.c_cc[VTIME] = 10;
200 
201  /* Enable local mode, because the USBD_VCOM APP does not handle the
202  * virtual flow control lines
203  */
204  options.c_cflag |= CLOCAL;
205 
206  /* Cause the new options to take effect immediately. */
207  if (tcsetattr(com_port_handle, TCSANOW, &options) == -1)
208  {
209  close(com_port_handle);
210  return -1;
211  }
212 
213  /* add the handle to the array of open COM ports */
214  /* --------------------------------------------- */
215 
216  /* increase capacity of handle array if needed */
217  if (num_open_handles == num_allocated_handles)
218  {
219  int* new_handles = (int*)malloc((num_open_handles + 1) * sizeof(int));
220 
221  if (num_allocated_handles > 0)
222  {
223  memcpy(new_handles, handles, sizeof(int) * num_allocated_handles);
224  }
225 
226  new_handles[num_allocated_handles++] = -1;
227 
228  free(handles);
229  handles = new_handles;
230  }
231 
232  /* add new handle to table */
233  new_handle = 0;
234 
235  while (handles[new_handle] != -1)
236  {
237  ++new_handle;
238  }
239 
240  handles[new_handle] = com_port_handle;
241  ++num_open_handles;
242 
243  return new_handle;
244 }

246 void com_close(int32_t port_handle)
247 {
248  /* check if handle is valid */
249  if ((port_handle >= 0) && (port_handle < num_allocated_handles) &&
250  (handles[port_handle] != -1))
251  {
252  /* close COM port */
253  close(handles[port_handle]);
254 
255  /* remove handle from the table */
256  handles[port_handle] = -1;
257  --num_open_handles;
258 
259  /* if all handles have been closed, free handle table to prevent
260  * memory leaks
261  */
262  if (num_open_handles == 0)
263  {
264  num_allocated_handles = 0;
265  free(handles);
266  handles = NULL;
267  }
268  }

282 size_t com_get_data(int32_t port_handle,
283  void* data, size_t num_requested_bytes)
284 {
285  /* check if handle is valid */
286  if ((port_handle >= 0) && (port_handle < num_allocated_handles) &&
287  (handles[port_handle] != -1))
288  {
289  size_t num_received_bytes = 0;
290  char* read_buffer = (char*)data;
291 
292  /* read data */
293  while (num_received_bytes < num_requested_bytes)
294  {
295  ssize_t num_bytes = read(handles[port_handle],
296  read_buffer + num_received_bytes,
297  num_requested_bytes -
298  num_received_bytes);
299 
300  if (num_bytes <= 0)
301  {
302  break;
303  }
304 
305  num_received_bytes += num_bytes;
306  }
307 
308  return num_received_bytes;
309  }
310 
311  /* if handle was not valid, return 0 to indicate that no data was
312  * received
313  */
314  return 0;
315 }

317 void com_set_timeout(int32_t port_handle, uint32_t timeout_period_ms)
318 {
319  /* check if handle is valid */
320  if ((port_handle >= 0) &&
321  (port_handle < (int32_t)num_allocated_handles) &&
322  (handles[port_handle] != -1))
323  {
324  /* get current settings */
325  struct termios options;
326  tcgetattr(handles[port_handle], &options);
327 
328  /* apply new timeout value */
329  options.c_cc[VTIME] = timeout_period_ms / 100;
330  tcsetattr(handles[port_handle], TCSANOW, &options);
331  }
332 }
