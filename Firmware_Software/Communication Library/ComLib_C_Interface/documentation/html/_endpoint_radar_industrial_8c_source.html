167 static int32_t parse_duty_cycle_status(int32_t protocol_handle, uint8_t endpoint,
168  const uint8_t* payload, uint16_t payload_size)
169 {
170  if ((protocol_read_payload_uint8(payload, 0) == MSG_IS_ENABLE_DUTY_CYCLE) &&
171  (payload_size == 2))
172  {
173  if (callback_duty_cycle_status)
174  {
175  uint8_t duty_cycle_is_enabled;
176 
177  duty_cycle_is_enabled = protocol_read_payload_uint8(payload, 1);
178 
179  duty_cycle_is_enabled = (duty_cycle_is_enabled > 0 ? 1 : 0);
180 
181  /* send frame format to callback */
182  callback_duty_cycle_status(context_duty_cycle_status, protocol_handle,
183  endpoint, duty_cycle_is_enabled);
184  }
185  return 1;
186  }
187  return 0;
188 }

213 static void parse_payload(int32_t protocol_handle, uint8_t endpoint,
214  const uint8_t* payload, uint16_t payload_size)
215 {
216  /* try to parse payload for all supported message types, stop when parsing
217  * was successful
218  */
219  if (parse_bgt_lna_status(protocol_handle, endpoint,
220  payload, payload_size))
221  return;
222 
223  if (parse_duty_cycle_status(protocol_handle, endpoint,
224  payload, payload_size))
225  return;
226 }
