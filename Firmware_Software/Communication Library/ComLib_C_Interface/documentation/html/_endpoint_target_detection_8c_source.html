183 static void parse_payload(int32_t protocol_handle, uint8_t endpoint,
184  const uint8_t* payload, uint16_t payload_size)
185 {
186  /* try to parse payload for all supported message types, stop when parsing
187  * was successful
188  */
189  if (parse_dsp_settings(protocol_handle, endpoint, payload, payload_size))
190  return;
191 
192  if (parse_target_info(protocol_handle, endpoint, payload, payload_size))
193  return;
194 
195  if (parse_range_threshold(protocol_handle, endpoint, payload, payload_size))
196  return;
197 }

199 static int32_t parse_target_info(int32_t protocol_handle, uint8_t endpoint,
200  const uint8_t* payload,
201  uint16_t payload_size)
202 {
203  if ((protocol_read_payload_uint8(payload, 0) == MSG_GET_TARGETS) &&
204  (payload_size >= 2))
205  {
206  // callbacks for target info are not defined yet
207  if (callback_target_processing)
208  {
209  Target_Info_t* targets;
210  uint8_t i;
211 
212  uint8_t num_targets = protocol_read_payload_uint8(payload, 1);
213 
214  const uint16_t target_data_size = sizeof(float) * 7 +
215  sizeof(uint32_t);
216 
217  if (payload_size == num_targets * target_data_size + 2)
218  {
219  targets = (Target_Info_t*)malloc(num_targets *
220  sizeof(Target_Info_t));
221 
222  for (i = 0; i < num_targets; ++i)
223  {
224  targets[i] = *(Target_Info_t*)(payload + 2 +
225  i * target_data_size);
226  }
227 
228  /* send target info to callback */
229  callback_target_processing(context_target_processing,
230  protocol_handle, endpoint,
231  targets, num_targets);
232 
233  free (targets);
234  }
235  }
236  return 1;
237  }
238  return 0;
239 }

241 static int32_t parse_dsp_settings(int32_t protocol_handle, uint8_t endpoint,
242  const uint8_t* payload, uint16_t payload_size)
243 {
244  if ((protocol_read_payload_uint8(payload, 0) == MSG_GET_DSP_SETTINGS))
245  {
246  if (callback_dsp_settings)
247  {
248  DSP_Settings_t dsp_settings;
249 
250  dsp_settings.range_mvg_avg_length = *(uint8_t*) (payload + 1);
251 
252  dsp_settings.min_range_cm = *(uint16_t*)(payload + 2);
253  dsp_settings.max_range_cm = *(uint16_t*)(payload + 4);
254 
255  dsp_settings.min_speed_kmh = *(uint16_t*)(payload + 6);
256  dsp_settings.max_speed_kmh = *(uint16_t*)(payload + 8);
257 
258  dsp_settings.min_angle_degree = *(uint16_t*)(payload + 10);
259  dsp_settings.max_angle_degree = *(uint16_t*)(payload + 12);
260 
261  dsp_settings.range_threshold = *(uint16_t*)(payload + 14);
262  dsp_settings.speed_threshold = *(uint16_t*)(payload + 16);
263 
264  if(payload_size == 27 )
265  {
266  // skip uint16 entry for index 18, see MMWSW-603
267  dsp_settings.enable_tracking = *(uint8_t*) (payload + 20);
268  dsp_settings.num_of_tracks = *(uint8_t*) (payload + 21);
269 
270  dsp_settings.median_filter_length = *(uint8_t*) (payload + 22);
271  dsp_settings.enable_mti_filter = *(uint8_t*) (payload + 23);
272 
273  dsp_settings.mti_filter_length = *(uint16_t*)(payload + 24);
274  // skip uint8 entry for index 26, see MMWSW-603
275  }
276  else // Filled with default values, instead of reading the garbage memory
277  {
278  dsp_settings.enable_tracking = 0;
279  dsp_settings.num_of_tracks = 1;
280 
281  dsp_settings.median_filter_length = 5;
282  dsp_settings.enable_mti_filter = 0;
283 
284  dsp_settings.mti_filter_length = 10;
285  }
286 
287 
288  /* send DSP settings to callback */
289  callback_dsp_settings(context_dsp_settings,
290  protocol_handle, endpoint,
291  &dsp_settings);
292  }
293  return 1;
294  }
295  return 0;
296 }

298 static int32_t parse_range_threshold(int32_t protocol_handle, uint8_t endpoint,
299  const uint8_t* payload, uint16_t payload_size)
300 {
301  if ((protocol_read_payload_uint8(payload, 0) == MSG_GET_RANGE_THRESHOLD) &&
302  (payload_size == 3))
303  {
304  if (callback_range_threshold)
305  {
306  uint16_t threshold;
307 
308  threshold = *(uint16_t*) (payload + 1);
309 
310  /* send DSP settings to callback */
311  callback_range_threshold(context_range_threshold,
312  protocol_handle, endpoint,
313  threshold);
314  }
315  return 1;
316  }
317  return 0;
318 }

367 
368  protocol_write_payload_uint16(cmd_message, 2, dsp_settings->min_range_cm);
369  protocol_write_payload_uint16(cmd_message, 4, dsp_settings->max_range_cm);
370  protocol_write_payload_uint16(cmd_message, 6, dsp_settings->min_speed_kmh);
371  protocol_write_payload_uint16(cmd_message, 8, dsp_settings->max_speed_kmh);
372  protocol_write_payload_uint16(cmd_message, 10, dsp_settings->min_angle_degree);
373  protocol_write_payload_uint16(cmd_message, 12, dsp_settings->max_angle_degree);
374  protocol_write_payload_uint16(cmd_message, 14, dsp_settings->range_threshold);
375  protocol_write_payload_uint16(cmd_message, 16, dsp_settings->speed_threshold);
376 
377  protocol_write_payload_uint16(cmd_message, 18, 0 );
378  protocol_write_payload_uint8 (cmd_message, 20, dsp_settings->enable_tracking);
379  protocol_write_payload_uint8 (cmd_message, 21, dsp_settings->num_of_tracks);
380 
381  protocol_write_payload_uint8 (cmd_message, 22, dsp_settings->median_filter_length);
382  protocol_write_payload_uint8 (cmd_message, 23, dsp_settings->enable_mti_filter);
383 
384  protocol_write_payload_uint16(cmd_message, 24, dsp_settings->mti_filter_length);
385  protocol_write_payload_uint16(cmd_message, 26, 0 );
386 
387 
388  /* send message and process received response */
389  return protocol_send_and_receive(protocol_handle, endpoint,
390  &ep_target_detection_definition,
391  cmd_message, sizeof(cmd_message));
