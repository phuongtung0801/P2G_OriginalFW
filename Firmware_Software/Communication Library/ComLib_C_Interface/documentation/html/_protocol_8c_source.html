308 static void recover_from_receive_error(int32_t com_port_handle)
309 {
310  /* read until buffer is empty */
311  uint8_t dummy_data[1024];
312  size_t received_bytes = sizeof(dummy_data);
313 
314  while (received_bytes == sizeof(dummy_data))
315  {
316  received_bytes = com_get_data(com_port_handle,
317  dummy_data, sizeof(dummy_data));
318  }
319 
320  /* now we have run out of data, protocol should be in sync again */
321 }

323 static void send_message(int32_t com_port_handle, uint8_t endpoint,
324  const uint8_t* payload, uint16_t payload_size)
325 {
326  /* setup message header and tail */
327  uint8_t message_header[4];
328  uint8_t message_tail[2];
329 
330  message_header[0] = CNST_STARTBYTE_DATA;
331  message_header[1] = endpoint;
332  message_header[2] = payload_size & 0xFF;
333  message_header[3] = payload_size >> 8;
334 
335  message_tail[0] = CNST_END_OF_PAYLOAD & 0xFF;
336  message_tail[1] = CNST_END_OF_PAYLOAD >> 8;
337 
338  /* send message */
339  com_send_data(com_port_handle, message_header, 4);
340  com_send_data(com_port_handle, payload, payload_size);
341  com_send_data(com_port_handle, message_tail, 2);
342 }

344 static int32_t get_message(int32_t com_port_handle,
345  Message_Info_t* message_info)
346 {
347  uint8_t message_header[4];
348  size_t num_received_bytes;
349 
350  /* read message header */
351  /* ------------------- */
352  num_received_bytes = com_get_data(com_port_handle,
353  message_header, sizeof(message_header));
354 
355  /*
356  * If number of received bytes is smaller than expected, one reason could
357  * be that the message came in shortly before the timeout. If this has
358  * happened, the rest of the message bytes must be available now. Try
359  * another read.
360  */
361  if ((num_received_bytes < sizeof(message_header)))
362  {
363  num_received_bytes += com_get_data(com_port_handle,
364  message_header +
365  num_received_bytes,
366  sizeof(message_header) -
367  num_received_bytes);
368  }
369 
370  /* if no data at all has been received, everything is all right, but there
371  * is no message
372  */
373  if (num_received_bytes == 0)
374  {
375  return PROTOCOL_ERROR_RECEIVED_NO_MESSAGE;
376  }
377 
378  /* if not enough bytes for a full message header have been received,
379  * something went wrong
380  */
381  else if (num_received_bytes < sizeof(message_header))
382  {
383  recover_from_receive_error(com_port_handle);
384  return PROTOCOL_ERROR_RECEIVED_TIMEOUT;
385  }
386 
387  /* read rest of message */
388  /* -------------------- */
389  if (message_header[0] == CNST_STARTBYTE_DATA)
390  {
391  uint8_t* payload;
392  uint16_t payload_size;
393  uint8_t message_tail[2];
394 
395  /* allocate memory and receive payload */
396  payload_size = (uint16_t)message_header[2] |
397  ((uint16_t)message_header[3]) << 8;
398  payload = (uint8_t*)malloc(payload_size);
399 
400  num_received_bytes = com_get_data(com_port_handle,
401  payload, payload_size);
402 
403  /* check if payload has been received completely */
404  if (num_received_bytes < payload_size)
405  {
406  /* free payload memory, it's incomplete */
407  free(payload);
408 
409  recover_from_receive_error(com_port_handle);
410 
411  return PROTOCOL_ERROR_RECEIVED_TIMEOUT;
412  }
413 
414  /* check message tail */
415  num_received_bytes = com_get_data(com_port_handle, &message_tail,
416  sizeof(message_tail));
417 
418  if ((num_received_bytes != sizeof(message_tail)) ||
419  (message_tail[0] != (CNST_END_OF_PAYLOAD & 0xFF)) ||
420  (message_tail[1] != CNST_END_OF_PAYLOAD >> 8))
421  {
422  /* free payload memory, it's incomplete */
423  free(payload);
424 
425  recover_from_receive_error(com_port_handle);
426 
427  return PROTOCOL_ERROR_RECEIVED_BAD_MESSAGE_END;
428  }
429 
430  message_info->endpoint = message_header[1];
431  message_info->payload = payload;
432  message_info->payload_size = payload_size;
433 
434  return CNST_PROTOCOL_RECEIVED_PAYLOAD_MSG;
435  }
436  else if (message_header[0] == CNST_STARTBYTE_STATUS)
437  {
438  uint8_t endpoint = message_header[1];
439  uint16_t status_code = (uint16_t)message_header[2] |
440  ((uint16_t)message_header[3]) << 8;
441 
442  return ((int)endpoint << 16) | status_code;
443  }
444  else
445  {
446  recover_from_receive_error(com_port_handle);
447  return PROTOCOL_ERROR_RECEIVED_BAD_MESSAGE_START;
448  }
449 }

457 int32_t protocol_connect(const char* port_name)
458 {
459  Instance_t protocol_instance;
460  Message_Info_t message_info;
461  int32_t receive_status;
462  uint8_t i;
463  int32_t new_handle;
464 
465  /* open the COM port */
466  /* ----------------- */
467  protocol_instance.com_port_handle = com_open(port_name);
468 
469  if (protocol_instance.com_port_handle < 0)
470  {
471  return PROTOCOL_ERROR_COULD_NOT_OPEN_COM_PORT;
472  }
473 
474  /* get Endpoint information from device */
475  /* ------------------------------------ */
476  protocol_instance.num_endpoints = 0;
477  protocol_instance.endpoints = NULL;
478 
479  /* send a message with command code to query endpoint info to endpoint 0
480  */
481  {
482  uint8_t uQueryMessage[1] = { CNST_MSG_QUERY_ENDPOINT_INFO };
483  send_message(protocol_instance.com_port_handle, 0,
484  uQueryMessage, sizeof(uQueryMessage));
485  }
486 
487  /* read and parse reply message from connected device */
488  /* -------------------------------------------------- */
489  message_info.payload = NULL;
490  receive_status = get_message(protocol_instance.com_port_handle,
491  &message_info);
492 
493  if ((receive_status != CNST_PROTOCOL_RECEIVED_PAYLOAD_MSG) ||
494  (message_info.endpoint != 0) ||
495  (message_info.payload_size < 2) ||
496  (message_info.payload[0] != CNST_MSG_ENDPOINT_INFO))
497  {
498  /* This is not the expected payload, clean up and quit. */
499  free(message_info.payload);
500  com_close(protocol_instance.com_port_handle);
501  return PROTOCOL_ERROR_DEVICE_NOT_COMPATIBLE;
502  }
503 
504  /* read number of endpoints and check message size */
505  protocol_instance.num_endpoints = message_info.payload[1];
506 
507  if ((message_info.payload_size !=
508  6 * protocol_instance.num_endpoints + 2) ||
509  (protocol_instance.num_endpoints == 0))
510  {
511  free(message_info.payload);
512  com_close(protocol_instance.com_port_handle);
513  return PROTOCOL_ERROR_DEVICE_NOT_COMPATIBLE;
514  }
515 
516  /* allocate array to hold endpoint information */
517  protocol_instance.endpoints =
518  (Endpoint_t*)malloc(protocol_instance.num_endpoints *
519  sizeof(Endpoint_t));
520 
521  /* iterate over all endpoint info records in the message */
522  for (i = 0; i < protocol_instance.num_endpoints; ++i)
523  {
524  uint32_t j;
525  Endpoint_t* endpoint = &protocol_instance.endpoints[i];
526 
527  /* read endpoint type and version from payload */
528  endpoint->type = protocol_read_payload_uint32(message_info.payload,
529  2 + i * 6);
530  endpoint->version = protocol_read_payload_uint16(message_info.payload,
531  6 + i * 6);
532 
533  /* find endpoint implementation matching type and version */
534  endpoint->endpoint_definiton = NULL;
535 
536  for (j = 0; j < NUM_KNOWN_ENDPOINTS; ++j)
537  {
538  if ((known_endpoints[j]->type == endpoint->type) &&
539  (known_endpoints[j]->min_version <= endpoint->version) &&
540  (known_endpoints[j]->max_version >= endpoint->version))
541  {
542  endpoint->endpoint_definiton = known_endpoints[j];
543  break;
544  }
545  }
546  }
547 
548  /* free payload memory */
549  free(message_info.payload);
550 
551  /* consume the expected status message */
552  receive_status = get_message(protocol_instance.com_port_handle,
553  &message_info);
554 
555  if (receive_status != ((/*endpoint*/0 << 16) | /*status code*/0x0000))
556  {
557  /*
558  * If the next message was not a status message with Code 0, something
559  * went wrong, so remove all the endpoints, close interface and
560  * return.
561  */
562  if (receive_status == CNST_PROTOCOL_RECEIVED_PAYLOAD_MSG)
563  {
564  free(message_info.payload);
565  }
566 
567  free(protocol_instance.endpoints);
568  com_close(protocol_instance.com_port_handle);
569  return PROTOCOL_ERROR_DEVICE_NOT_COMPATIBLE;
570  }
571 
572  /* add the handle to the array of open COM ports */
573  /* --------------------------------------------- */
574 
575  /* increase capacity of handle array if needed */
576  if (num_open_handles == num_allocated_handles)
577  {
578  Instance_t* pNewHandles = (Instance_t*)malloc((num_open_handles + 1) *
579  sizeof(Instance_t));
580 
581  if (num_allocated_handles > 0)
582  {
583  memcpy(pNewHandles, handles,
584  sizeof(Instance_t) * num_allocated_handles);
585  }
586 
587  pNewHandles[num_allocated_handles++].num_endpoints = 0;
588 
589  free(handles);
590  handles = pNewHandles;
591  }
592 
593  /* add new handle to table */
594  new_handle = 0;
595 
596  while (handles[new_handle].num_endpoints != 0)
597  {
598  ++new_handle;
599  }
600 
601  handles[new_handle] = protocol_instance;
602  ++num_open_handles;
603 
604  return new_handle;
605 }

607 void protocol_disconnect(int32_t protocol_handle)
608 {
609  /* check if handle is valid */
610  if ((protocol_handle >= 0) &&
611  (protocol_handle < (int32_t)num_allocated_handles) &&
612  (handles[protocol_handle].num_endpoints != 0))
613  {
614  /* close COM port */
615  com_close(handles[protocol_handle].com_port_handle);
616  handles[protocol_handle].com_port_handle = -1;
617 
618  /* free memory of endpoint table */
619  free (handles[protocol_handle].endpoints);
620  handles[protocol_handle].endpoints = NULL;
621 
622  /* remove handle from the table */
623  handles[protocol_handle].num_endpoints = 0;
624  --num_open_handles;
625 
626  /* if all handles have been closed, free handle table to prevent
627  * memory leaks
628  */
629  if (num_open_handles == 0)
630  {
631  num_allocated_handles = 0;
632  free(handles);
633  handles = NULL;
634  }
635  }

910  int32_t status_code)

913  if ((status_code & 0xFFFF) == PROTOCOL_STATUS_OK)
