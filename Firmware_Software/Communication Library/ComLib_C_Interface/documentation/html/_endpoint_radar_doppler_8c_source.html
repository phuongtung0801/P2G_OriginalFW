167 static int32_t parse_doppler_configuration(int32_t protocol_handle,
168  uint8_t endpoint,
169  const uint8_t* payload,
170  uint16_t payload_size)
171 {
172  if ((protocol_read_payload_uint8(payload, 0) == MSG_SET_CONFIGURATION) &&
173  (payload_size == 6))
174  {
175  if (callback_doppler_configuration)
176  {
177  Doppler_Configuration_t doppler_configuration;
178 
179  doppler_configuration.frequency_kHz =
180  protocol_read_payload_uint32(payload, 1);
181  doppler_configuration.tx_power =
182  protocol_read_payload_uint8 (payload, 5);
183 
184  /* send RF Configuration to callback */
185  callback_doppler_configuration(context_doppler_configuration,
186  protocol_handle, endpoint,
187  &doppler_configuration);
188  }
189 
190  return 1;
191  }
192 
193  return 0;
194 }
