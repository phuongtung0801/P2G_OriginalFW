344 static int32_t parse_frame_info(int32_t protocol_handle, uint8_t endpoint,
345  const uint8_t* payload,
346  uint16_t payload_size)
347 {
348  const uint16_t header_size = 18;
349 
350  if ((protocol_read_payload_uint8(payload, 0) == MSG_FRAME_DATA) &&
351  (payload_size >= header_size))
352  {
353  if (callback_data_frame)
354  {
355  Frame_Info_t frame_info;
356  uint32_t total_samples;
357  uint32_t expected_message_size;
358 
359  /* read header info from payload */
360  frame_info.frame_number =
361  protocol_read_payload_uint32(payload, 1);
362  frame_info.num_chirps = protocol_read_payload_uint32(payload, 5);
363  frame_info.num_rx_antennas =
364  protocol_read_payload_uint8 (payload, 9);
365  frame_info.num_samples_per_chirp =
366  protocol_read_payload_uint32(payload, 10);
367  frame_info.rx_mask = protocol_read_payload_uint8 (payload, 14);
368  frame_info.data_format =
369  (Rx_Data_Format_t)protocol_read_payload_uint8 (payload, 15);
370  frame_info.adc_resolution =
371  protocol_read_payload_uint8 (payload, 16);
372  frame_info.interleaved_rx =
373  protocol_read_payload_uint8 (payload, 17);
374 
375  /* calculate expected message size */
376  total_samples = frame_info.num_chirps *
377  frame_info.num_samples_per_chirp *
378  frame_info.num_rx_antennas *
379  (frame_info.data_format == EP_RADAR_BASE_RX_DATA_REAL ? 1 : 2);
380 
381  expected_message_size = total_samples * frame_info.adc_resolution;
382  expected_message_size = header_size +
383  (expected_message_size >> 3) +
384  ((expected_message_size & 0x07) ? 1 : 0);
385 
386  if (payload_size == expected_message_size)
387  {
388  uint32_t sample_bit_mask = (1 << frame_info.adc_resolution)
389  - 1;
390  const float norm_factor = 1.f / (float)sample_bit_mask;
391  float* sample_data = (float*)malloc(total_samples *
392  sizeof(float));
393  float* write_ptr = sample_data;
394 
395  uint16_t read_idx = header_size;
396  uint16_t read_bit_position = 0;
397 
398  while (total_samples)
399  {
400  uint32_t current_sample =
401  protocol_read_payload_uint16(payload, read_idx) >>
402  read_bit_position;
403  *write_ptr++ = (current_sample & sample_bit_mask) *
404  norm_factor;
405 
406  read_bit_position += frame_info.adc_resolution;
407  read_idx += read_bit_position >> 3;
408  read_bit_position &= 0x07;
409 
410  --total_samples;
411  }
412 
413  /* send frame info to callback */
414  frame_info.sample_data = sample_data;
415  callback_data_frame(context_data_frame, protocol_handle,
416  endpoint, &frame_info);
417  free(sample_data);
418  }
419 
420  return 1;
421  }
422  }
423 
424  return 0;
425 }
